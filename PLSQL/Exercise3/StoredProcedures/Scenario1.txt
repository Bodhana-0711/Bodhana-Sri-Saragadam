DELIMITER //
CREATE PROCEDURE ProcessMonthlyInterest()
BEGIN
    UPDATE Accounts
    SET balance = balance + (balance * 0.01)
    WHERE account_type = 'savings';

    SELECT 'Monthly interest applied successfully' AS message;
END//
DELIMITER ;
CALL ProcessMonthlyInterest();
