DELIMITER //
CREATE PROCEDURE UpdateEmployeeBonus(
    IN dept VARCHAR(50),
    IN bonus_percent INT
)
BEGIN
    UPDATE Employees
    SET salary = salary + (salary * bonus_percent / 100)
    WHERE department = dept;

    SELECT 'Bonus updated successfully' AS message;
END//
DELIMITER ;
CALL UpdateEmployeeBonus('IT', 10);
